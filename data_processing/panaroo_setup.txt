# Log into the remote Ubuntu instance
# ssh -i ~/.ssh/panaroo-key.pem ubuntu@ip

# Install Miniconda3
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
bash Miniconda3-latest-Linux-x86_64.sh -b -p $HOME/miniconda
export PATH="$HOME/miniconda/bin:$PATH"
source "$(conda info --base)/etc/profile.d/conda.sh"

# Create isolated Conda environment for Panaroo
conda create -n panaroo_env python=3.10 biopython=1.80 panaroo mafft -c conda-forge -c bioconda -y
conda activate panaroo_env
pip install gdown

# Prepare directories and download Prokka GFF annotation files
mkdir -p data/processed/prokka_out
gdown "https://drive.google.com/....=sharing"
gdown "https://drive.google.com/....=sharing"

# Extract GFF annotation files and clean zip files
unzip '*.zip' -d data/processed/prokka_out/
rm *.zip

# Flatten directory structures
mkdir -p data/processed/panaroo_input
find data/processed/prokka_out -name "*.gff" -exec cp {} data/processed/panaroo_input/ \;
rm -rf data/processed/prokka_out/

# Data Quality Control and Cleaning
find data/processed/panaroo_input -name "*.gff" -size 0 -delete
find data/processed/panaroo_input -name "*.gff" | xargs -I {} bash -c "grep -q $'\t' {} || echo {}" > empty_gffs.txt
if [ -s empty_gffs.txt ]; then 
    cat empty_gffs.txt | xargs rm
fi

# Purge gene count outliers (Standard K. pneumoniae limit: 4500-6500 CDS)
for gff in data/processed/panaroo_input/*.gff; do
    count=$(grep -c -P '\tCDS\t' "$gff")
    if [ "$count" -lt 4500 ] || [ "$count" -gt 6500 ]; then
        echo "Removing outlier: $(basename "$gff") (CDS Count: $count)"
        rm "$gff"
    fi
done

# Create run_panaroo.sh script
cat << 'EOF' > run_panaroo.sh
#!/bin/bash
export PATH="$HOME/miniconda/bin:$PATH"
source "$(conda info --base)/etc/profile.d/conda.sh"
conda activate panaroo_env

INPUT_DIR="data/processed/panaroo_input"
OUT_DIR="data/processed/panaroo_out"
LOG_DIR="logs"
export TMPDIR="$PWD/data/tmp"
mkdir -p "$OUT_DIR" "$LOG_DIR" "$TMPDIR"

# Execute Panaroo ( r5.8xlarge instance: 32 Cores )
nohup panaroo \
    -i "$INPUT_DIR"/*.gff \
    -o "$OUT_DIR" \
    --clean-mode strict \
    --alignment core \
    --aligner mafft \
    --core_threshold 0.95 \
    --len_dif_percent 0.95 \
    -c 0.98 \
    -t 32 \
    > "$LOG_DIR/panaroo_run.log" 2>&1 &
EOF

# Execute run_panaroo.sh
chmod +x run_panaroo.sh
./run_panaroo.sh
tail -f logs/panaroo_run.log

# Zip and archive results
cd data/processed/panaroo_out/
cp ~/logs/panaroo_run.log ./full_process_history.log
zip -r full_pangenome_results.zip \
    gene_presence_absence.csv \
    core_gene_alignment_filtered.aln \
    final_graph.gml \
    summary_statistics.txt \
    full_process_history.log \
    struct_presence_absence.Rtab \
    pan_genome_reference.fa \
    alignment_entropy.csv \
    pre_filt_graph.gml

# Start file transfer server
python3 -m http.server 8000

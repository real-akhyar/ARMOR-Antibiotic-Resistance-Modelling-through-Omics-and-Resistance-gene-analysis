# Process termination: stop existing RGI and Snippy runs
pkill -f run_rgi_parallel.py
pkill -f rgi
pkill -u ubuntu -f "python3 run_rgi_parallel.py"
pkill -u ubuntu -f "run_snippy_parallel.sh"
pkill -u ubuntu -f "snippy"
pkill -u ubuntu -f "xargs"

# Background execution: run parallel runs in background
nohup python3 run_rgi_parallel.py > logs/rgi_parallel.log 2>&1 & disown
chmod +x run_snippy_parallel.sh
nohup ./run_snippy_parallel.sh > logs/snippy_parallel.log 2>&1 & disown

# Stream monitoring: tail execution logs
tail -f logs/rgi_parallel.log
tail -f logs/snippy_parallel.log
